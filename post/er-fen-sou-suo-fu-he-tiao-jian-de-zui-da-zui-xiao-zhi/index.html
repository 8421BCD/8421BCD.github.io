二分搜索符合条件的最大最小值

发布于 2020-02-04 | 分类于 二分 | 3分钟 | 434字数

当要求解的问题有多种解时，要求符合条件的最大/小的值，可以通过二分枚举解来解决（同时注意二分过程的判断和终值，可以作为模板背过）

poj 2456 Aggressive cows | 求解符合条件值的最大值

题意：
给你n个坐标，求c个坐标中最小的最大距离。
分析：
注意while()循环的条件是l <= r， 最后结果是r。
同时判断函数可以通过贪心的思想写。
代码：

#include <iostream>
#include <algorithm>
using namespace std;
const int maxn = 1e5 + 10;
int v[maxn];
int n, x;

int judge(int dis)
{
	int lastpos = 0;
	int cnt = 1;
	for(int i = 1; i < n; i++)
		if(v[i] - v[lastpos] >= dis)
			lastpos = i, cnt++;
	return cnt;
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	while(cin >> n >> x)
	{
		for(int i = 0; i < n; i++)
		 	scanf("%d", v + i);
		sort(v, v + n);;
		// cout << judge(3) << endl;
		int l = 1, r = 1e9;
		while(l <= r)
		{
			int mid = (l + r) >> 1;
			if(judge(mid) >= x) l = mid + 1;
			else r = mid - 1;
		}
		cout << r << endl;
	}
}

Distributing Ballot Boxes（HDU1490）| 求解符合条件的最小值

题意：
给出n个城镇的人，m个投票箱，求出最小票箱容积，使得最快完成投票收集。
分析：
同上，最后返回l
代码：

#include <bits/stdc++.h>
using namespace std;
const int maxn = 5e5 + 10;
int n, m;
int v[maxn];

int judge(int maxx)
{
	int cnt = 0;
	for(int i = 1; i <= n; i++)
		if(v[i] % maxx == 0) cnt += v[i] / maxx;
		else cnt += v[i] / maxx + 1;
	return cnt;
}

int main()
{
	while(cin >> n >> m && n != -1)
	{
		for(int i = 1; i <= n; i++)
			scanf("%d", v + i);;
		int l = 1, r = 5e6;
		while(l <= r)
		{
			int mid = l + r >> 1;
			if(judge(mid) <= m) r = mid - 1;
			else l = mid + 1;
		}
		cout << l << endl;
	}
}
